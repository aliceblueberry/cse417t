List of mysql commands to create the necessary tables for the news site:

create table users (
    username VARCHAR(50) not null,
    hashed_password CHAR(50) not null,
    primary key (username, hashed_password)
) engine = InnoDB default character set = utf8 collate = utf8_general_ci;

create table comments (
    comment_id MEDIUMINT unsigned not null auto_increment,
    body TEXT not null,
    username VARCHAR(50) not null,
    story MEDIUMINT unsigned not null,
    primary key (comment_id),
    foreign key (username) references users (username),
    foreign key (story) references stories (story_id)
) engine = InnoDB default character set = utf8 collate = utf8_general_ci;



create table links (
    link_id MEDIUMINT unsigned not null auto_increment,
    link TEXT not null,
    username VARCHAR(50) not null,
    story MEDIUMINT unsigned not null,
    primary key (link_id),
    foreign key (username) references users (username),
    foreign key (story) references stories (story_id)
) engine = InnoDB default character set = utf8 collate = utf8_general_ci;

create table stories ( 
    story_id MEDIUMINT unsigned not null auto_increment,
    title TINYTEXT not null,
    body TEXT not null,
    username VARCHAR(50) not null,
    primary key (story_id),
    foreign key (username) references users (username)
) engine = InnoDB default character set = utf8 collate = utf8_general_ci;
